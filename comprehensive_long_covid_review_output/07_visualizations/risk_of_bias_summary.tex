    axis lines=left,
    grid=both,
    grid style={line width=.1pt, draw=gray!20},
    major grid style={line width=.2pt, draw=gray!40},
    enlarge x limits={abs=0.5},
    legend style={at={(0.02,0.98)},anchor=north west, font=\footnotesize},
    clip=false,
    ybar stacked,
    bar width=0.6cm,
